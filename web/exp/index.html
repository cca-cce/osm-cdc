Welcome to Random Page Redirect

from js import document, window import random def create_random_button(): btn = document.createElement('button') btn.textContent = 'Go to a random page' btn.onclick = redirect_user document.body.appendChild(btn) def redirect_user(event): window.location.href = 'subpage_a.html' if random.random() < 0.5 else 'subpage_b.html' create_random_button()
